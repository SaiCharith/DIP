my_display(im4,'original');

B = imhist(im6);

for k=1:size(im4,3)

    A = imhist(im4(:,:,k));

    A(1) = A(1) - B(1);
    A = cumsum(A);
    A = A/A(256);
    for i=1:size(im4,1)
       im(i,:,k) = A(im4(i,:,k)+1)*255;
    end


    x= im(:,:,k).*im6;
    Iy(:,:,k) = x;
end
